\section{Introduction}
In this note, we consider ways to implement the 
Coulomb matrix:
\begin{equation}
J_{\mu \nu} = \sum_{\rho \sigma} (\mu \nu | \rho \sigma) D_{\sigma \rho},
\end{equation}
and the exchange matrix:
\begin{equation}
K_{\mu \nu} = \sum_{\rho \sigma} (\mu \sigma | \rho \nu) D_{\sigma \rho},
\end{equation}
where we take the density matrix to be defined as
\begin{equation}
D_{\sigma \rho} = \sum_i^{\mathrm{occ}} n_i C_{\sigma i} C^*_{\rho i}.
\end{equation}

Let's consider an auxiliary basis and the corresponding approximation to the four-center Coulomb integrals
\begin{equation}
\begin{split}
(\mu \nu | \rho \sigma) & = \sum_{PQ} (\mu \nu | P) (P|Q)^{-1} (Q|\rho \sigma) \\
& = \sum_{PQR} (\mu \nu | P) (P|Q)^{-1/2} (Q|R)^{-1/2} (R|\rho \sigma) \\
& = \sum_{P} B^{P}_{\mu \nu} B^{P}_{\rho \sigma},
\end{split}
\end{equation}
where we have defined
\begin{equation}
B^{P}_{\mu \nu} = \sum_Q (P|Q)^{-1/2} (Q|\mu \nu).
\end{equation}

The matrix inverse square root of the Coulomb metric $A\equiv (P|Q)$ can be computed by Cholesky decomposition:
\begin{equation}
A = L L^T,
\end{equation}
the matrix inverse is 
\begin{equation}
\label{eq:inv}
A^{-1} = L^{-T} L^{-1},
\end{equation}
From \cref{eq:inv}, we can identify $L^{-1}$ as a valid choice for $A^{-1/2}$ in our context, since $L^{-1}_{PR}(R|\rho \sigma) = B^{P}_{\rho \sigma}$, and the contraction of the B tensors yields the original four-center integrals:
\begin{align}
\sum_{P} B^{P}_{\mu \nu} B^{P}_{\rho \sigma} 
& = \sum_{PQR}(\mu \nu | P) L^{-\text{T}}_{PQ} L^{-1}_{QR} (R|\rho \sigma) \\
& = \sum_{PQ}(\mu \nu | P) (P|Q)^{-1} (Q|\rho \sigma) \\
& = (\mu \nu | \rho \sigma).
\end{align}

\section{Coulomb matrix algorithm}

The Coulomb matrix can be computed efficiently as
\begin{equation}
\begin{split}
J_{\mu \nu} = \sum_{P} B^{P}_{\mu \nu}
\underbrace{\left( \sum_{\rho \sigma}B^{P}_{\rho \sigma} D_{\sigma \rho} \right)}_{b^P}=
\sum_{P} B^{P}_{\mu \nu} b^P.
\end{split}
\end{equation}
If the B tensors can be stored, then the contraction as written above can be directly implemented with \texttt{numpy.einsum} without further optimization, \textit{i.e.},
\begin{quote}
    \texttt{np.einsum("Pmn,Prs,sr->mn", B, B, D, optimize=True)}
\end{quote}
This contraction is cache-friendly since the contraction happens over the outer indices.

However, if we cannot store the B tensors, then we can do the following:
\begin{equation}
\begin{split}
J_{\mu \nu} & = \sum_{PQ} (\mu \nu | P) (P|Q)^{-1}
\underbrace{\sum_{\rho \sigma} (Q|\rho \sigma) D_{\sigma \rho}}_{a_Q} \\
 & = \sum_{P} (\mu \nu | P) \underbrace{\sum_{Q} (P|Q)^{-1} a_Q}_{b_P} \\
 & = \sum_{P} (\mu \nu | P) b_P
\end{split}
\end{equation}

The Coulomb operator can be implemented in the following way.
\begin{enumerate}
    \item Compute the vector of size $N_\mathrm{aux}$ on the fly without storage:
\begin{equation}
a_Q = \sum_{\rho \geq \sigma} (Q|\rho \sigma) D_{\sigma \rho} (1+\delta_{\rho \sigma})
\end{equation}
In this step, one can apply screening based on $D_{\sigma \rho}$ and $(Q|\rho \sigma)$. 

    \item Matrix vector multiplication
\begin{equation}
\label{eq:bp}
b_P = \sum_{Q} (P|Q)^{-1} a_Q
\end{equation}
    \item Second pass over all the three-center integrals
\begin{equation}
J_{\mu \nu} = \sum_{P} (\mu \nu | P) b_P, \quad \text{ for } \mu \geq \nu.
\end{equation}
In this step, one can apply integral screening of $(\mu \nu | P)$ to reduce the evaluation cost.
\end{enumerate}

Note in \cref{eq:bp}, we require the application of the inverse of the Coulomb metric to a vector.
There are at least four, in principle equivalent, ways to do this:
\begin{enumerate}
    \item Use the pre-computed $L^{-1}$ to form $A^{-1}=L^{-T} L^{-1}$ and do a matrix-vector product with the problem vector: $b_Q = A^{-1} a_Q$.
    \item Two triangular solves with $L$ on the problem vector $a_Q$: first solve $L z = a_Q$ for $z$ and then solve $L^T b_P = z$ for $b_P$.
    \item Compute the eigen-decomposition of the Coulomb metric $A = U s U^T$ and threshold the eigenvalues to get $A_{\eta}^{-1} = U_{\eta} s_{\eta}^{-1} U_{\eta}^T$ where $s_{\eta}$ are the eigenvalues above some threshold $\eta$ and $U_{\eta}$ are the corresponding eigenvectors. Then do a matrix-vector product with the problem vector: $b_P = A_{\eta}^{-1}a_Q$.
    \item Compute the same $U_{\eta}$ and $s_{\eta}$ as above, but apply it to each problem vector without forming $A_{\eta}^{-1}$: $b_P = U_{\eta} (U_{\eta}^T a_Q / s_{\eta})$.
\end{enumerate}
The first two methods cannot deal with Coulomb metrics that have large condition numbers. The first and third methods at first seem appealing, since we pre-compute a `universal' $A^{-1}$ and just do matrix-vector products later.
Turns out, forming of $A^{-1}$ directly, as in the first and third methods is not numerically stable, since the error of these operations is on the order of $\text{cond}(A)^2\epsilon$, whereas the error of the second and fourth methods is on the order of $\text{cond}(A)\epsilon$.

If ill-conditioning is not a problem, then \texttt{scipy} offers a very succint way to carry out method 2:
\begin{quote}
    \texttt{bP = scipy.linalg.cho\_solve((L, True), aQ)}
\end{quote}
which does the two triangular solves with the Cholesky factor $L$ in one call.

\section{Exchange matrix algorithm}

The exchange matrix can be computed efficiently as
\begin{equation}
\label{eq:exchange}
\begin{split}
K_{\mu \nu} &= \sum_{P} \sum_{i}
\sum_{\sigma \rho} B^{P}_{\mu \sigma} B^{P}_{\rho \nu} C_{\sigma i} C^*_{\rho i} n_i \\
&= \sum_{P} \sum_{i}
\underbrace{\left(\sum_{\sigma} B^{P}_{\mu \sigma}  C_{\sigma i}\sqrt{n_i} \right)}_{Y_{P \mu i}}
\left(\sum_{\rho} B^{P}_{\rho \nu} C^*_{\rho i} \sqrt{n_i}\right) \\
&= \sum_{P} \sum_{i}
Y_{P \mu i} Y^*_{P \nu i}
\end{split}
\end{equation}
If the B tensors can be stored, then, again, the contractions above can be directly implemented with \texttt{numpy.einsum}:
\begin{quote}
    \texttt{Y = np.einsum("Pms,si->Pmi", B, Ci, optimize=True)}\\
    \texttt{K = np.einsum("Pmi,Pni->mn", Y, Y.conj(), optimize=True)}
\end{quote}
These contractions however, are not cache-friendly since the K contraction happens over the first and third indices, making memory access strided.
A better contraction would have been
\begin{quote}
    \texttt{Y = np.einsum("nPr,ri->nPi", B, Ci.conj(), optimize=True)}\\
    \texttt{K = np.einsum("mPi,nPi->mn", Y.conj(), Y, optimize=True)}
\end{quote}
which requires the B tensors to be stored in a transposed (nPm) order.
Doing this greatly improves the cache performance (3-5x speedup) of the K contraction at the cost of doubling the memory requirements.

When the B tensors cannot be stored, we need to perform the contractions on the fly. 
The most expensive step is the evaluation of $Y_{P \mu i}$ which scales as $N_\mathrm{aux} N_{\mathrm{AO}}^2 N_{\mathrm{occ}}$ and requires additional storage of $N_\mathrm{aux} N_{\mathrm{AO}} N_{\mathrm{occ}}$.

\begin{equation}
\begin{split}
Y_{P \mu i} & = \sum_{\sigma} B^{P}_{\mu \sigma} C_{\sigma i}\sqrt{n_i} \\
& = \sum_{\sigma} \sum_{Q} (P|Q)^{-1/2} (Q | \mu \sigma) C_{\sigma i}\sqrt{n_i} \\
& = \sum_{Q} (P|Q)^{-1/2} \sum_{\sigma} (Q | \mu \sigma) C_{\sigma i}\sqrt{n_i} \\
& = \sum_{Q} (P|Q)^{-1/2} X_{Q \mu i},
\end{split}
\end{equation}
where
\begin{equation}
X_{Q \mu i} = \sum_{\sigma} (Q | \mu \sigma) C_{\sigma i}\sqrt{n_i}
\end{equation}

The exchange operator can be implemented in the following way.
\begin{enumerate}
    \item Compute the tensor $X_{Q \mu i}$ with storage cost of $N_\mathrm{aux} N_{\mathrm{AO}} N_{\mathrm{occ}}$ by doing one on-the-fly pass over the three-center integrals, evaluating only the part $\mu \geq \sigma$ and using the fact that $(Q | \mu \sigma)=(Q | \sigma \mu)$:
\begin{equation}
X_{Q \mu i} = \sum_{\sigma} (Q | \mu \sigma) C_{\sigma i} \sqrt{n_i}
\end{equation}
Here if the orbital is localized, then one can apply screening based on only those shells on which $C_{\sigma i}$ has support and this can be combined with integral screening of $(Q | \mu \sigma)$.

    \item Compute the matrix-tensor product which yields an intermediate of the same dimension ($N_\mathrm{aux} N_{\mathrm{AO}} N_{\mathrm{occ}}$):
\begin{equation}
Y_{P \mu i} = \sum_{Q} (P|Q)^{-1/2} X_{Q \mu i},
\end{equation}
    \item Contract the two tensors:
\begin{equation}
K_{\mu \nu} = \sum_{P} \sum_{i} Y_{P \mu i} Y^*_{P \nu i}
\end{equation}
\end{enumerate}
If we cannot store the full $X_{Q \mu i}$ then we can do the following. Split the sum over $i$ into
\begin{equation}
\sum_i = \sum_{\alpha=1}^{r} \sum_{i \in [i]_\alpha}
\end{equation}
where $[i]_\alpha$ indicates a range of indices $[i]_\alpha = \{n_{\alpha-1}, n_{\alpha-1} + 1, n_{\alpha-1} + 2, \ldots, n_{\alpha} - 1\}$ where $n_{\alpha-1}$ and $n_{\alpha}$ are the start and end (excluded) of the range.
Then we can split the ranges evenly and determine the maximum size of the slice of the tensor $X_{Q\mu [i]_\alpha}$ that fits in memory:
\begin{equation}
r = \frac{N_\mathrm{aux} N_{\mathrm{AO}} N_{\mathrm{occ}}}{\textrm{memory}}
\end{equation}

Then the algorithm is implemented by performing $r$ passes over the three-center integrals and storing
\begin{equation}
X_{Q \mu [i]_\alpha} = \sum_{\sigma} (Q | \mu \sigma) C_{\sigma [i]_\alpha}\sqrt{n_{[i]_\alpha}}
\end{equation}
which is then transformed to $Y$ and accumulated into the exchange matrix.
Since we need to do two passes for the Coulomb matrix, we can go ahead and use these two passes to compute both $J$ and $K$ at the same time reducing the storage cost of the exchange algorithm to $N_\mathrm{aux} N_{\mathrm{AO}} N_{\mathrm{occ}}/2$.

\section{Integral screening}

Consider the Taylor expansion of the Coulomb operator
\begin{equation}
\frac{1}{r} = \frac{1}{\sqrt{\mathbf{r}\cdot \mathbf{r}}}    
\end{equation}

\begin{equation}
\frac{1}{r} = \frac{1}{r_0}
+ \left. \frac{\partial}{\partial x}\frac{1}{r} \right|_{x = x_0} (x - x_0) + \cdots
+ \frac{1}{2}\left. \frac{\partial^2}{\partial x^2}\frac{1}{r} \right|_{x = x_0} (x - x_0)^2
+ \frac{1}{2}\left. \frac{\partial^2}{\partial x \partial y}\frac{1}{r} \right|_{x = x_0} (x - x_0)(y - y_0) + \cdots.
\end{equation}
We have 
\begin{equation}
\left. \frac{\partial}{\partial x}\frac{1}{r} \right|_{x = x_0}
=    
\left. \frac{\partial}{\partial x}(x^2 + y^2 + z^2)^{-1/2} \right|_{x = x_0}
=  -  
\left. \frac{1}{2}(x^2 + y^2 + z^2)^{-3/2} 2 x \right|_{x = x_0}
=-\frac{1}{r_0^3} x_0
\end{equation}
We have 
\begin{equation}
\begin{split}
\left. \frac{\partial^2}{\partial x^2}\frac{1}{r} \right|_{x = x_0}
& = -
\left. \frac{\partial}{\partial x}(x^2 + y^2 + z^2)^{-3/2} x_0 \right|_{x = x_0} \\
& = \frac{3}{2}\left. (x^2 + y^2 + z^2)^{-5/2} 2 x^2_0 \right|_{x = x_0}
 -
\left. (x^2 + y^2 + z^2)^{-3/2} \right|_{x = x_0} \\
&= \frac{3}{r_0^5} x^2 - \frac{1}{r_0^3}
\end{split}
\end{equation}


\begin{equation}
\frac{1}{r} = \frac{1}{r_0} - \frac{1}{r_0^3} \mathbf{r}_0 \cdot (\mathbf{r} - \mathbf{r}_0) + \cdots
=\frac{1}{r_0} - \frac{\mathbf{r}_0 \cdot \mathbf{r}}{r_0^3}
+ \frac{1}{r_0} + \cdots
= \frac{2}{r_0} - \frac{\mathbf{r}_0 \cdot \mathbf{r}}{r_0^3}
 + \cdots 
\end{equation}
